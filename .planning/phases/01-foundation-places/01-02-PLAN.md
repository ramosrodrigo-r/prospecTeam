```javascript
export function parseArgs(argv) {
  const args = {}
  for (let i = 2; i < argv.length; i++) {
    if (argv[i] === '--city' && i + 1 < argv.length) {
      args.city = argv[++i]
    } else if (argv[i] === '--category' && i + 1 < argv.length) {
      args.category = argv[++i]
    }
  }
  return args
}
```

Export `searchPlaces({ query, pageToken, apiKey })` that makes one POST request to the Places API:

```javascript
export async function searchPlaces({ query, pageToken = null, apiKey }) {
  const headers = {
    'Content-Type': 'application/json',
    'X-Goog-Api-Key': apiKey,
    'X-Goog-FieldMask': 'places.id,places.displayName,places.websiteUri,places.nationalPhoneNumber,places.rating,nextPageToken'
  }

  const body = { textQuery: query, pageSize: 20 }
  if (pageToken) body.pageToken = pageToken

  if (process.env.DEBUG === '1') {
    console.error('[DEBUG] Request headers:', JSON.stringify(headers, null, 2))
    console.error('[DEBUG] Request body:', JSON.stringify(body, null, 2))
  }

  const response = await fetch('https://places.googleapis.com/v1/places:searchText', {
    method: 'POST',
    headers,
    body: JSON.stringify(body)
  })

  if (!response.ok) {
    const err = await response.text()
    throw new Error(`Places API error ${response.status}: ${err}`)
  }

  return response.json()
}
```

CRITICAL details:
- FieldMask header value is EXACTLY: `places.id,places.displayName,places.websiteUri,places.nationalPhoneNumber,places.rating,nextPageToken`
- Note: `nextPageToken` is top-level (NOT prefixed with `places.`) — this is the #1 pitfall from research
- `pageToken` is only included in body when truthy (not when null/undefined)
- DEBUG logging happens BEFORE the fetch call, uses `console.error` (not console.log)
- URL is exactly `https://places.googleapis.com/v1/places:searchText`
- Method is `POST`
- `pageSize` is always `20`

Create `src/stages/fetch.js` that orchestrates pagination over the Places API (per D-05, D-06).

```javascript
import { searchPlaces } from '../services/places.js'

const sleep = (ms) => new Promise(resolve => setTimeout(resolve, ms))

export async function fetchProspects({ city, category, apiKey }) {
  const query = `${category} em ${city}`
  const results = []
  let pageToken = null

  do {
    if (pageToken) await sleep(2500)

    const data = await searchPlaces({ query, pageToken, apiKey })

    for (const place of (data.places ?? [])) {
      results.push({
        placeId: place.id,
        name: place.displayName?.text ?? null,
        rating: place.rating ?? null,
        phone: place.nationalPhoneNumber ?? null,
        website: place.websiteUri ?? null,
        email: null
      })
    }

    pageToken = data.nextPageToken ?? null
  } while (pageToken)

  return results
}
```

CRITICAL details:
- Query format is `"${category} em ${city}"` (per D-04) — Portuguese preposition "em"
- Sleep 2500ms BEFORE using nextPageToken (per D-05), NOT after
- `place.displayName?.text` — displayName is a `{ text, languageCode }` object, NOT a string (Pitfall #2)
- All optional fields use `?? null` fallback
- `email` is hardcoded to `null` in Phase 1 (per D-10)
- `data.places ?? []` handles empty responses gracefully
- `data.nextPageToken ?? null` — if absent, loop terminates

```javascript
  process.exit(1)
}

// 2. Parse CLI args (per D-11, D-12)
const { city, category } = parseArgs(process.argv)
if (!city || !category) {
  console.error('Usage: node bin/prospect.js --city "Sao Paulo" --category "restaurante"')
  process.exit(1)
}

// 3. Fetch prospects (per D-05, D-06, D-09)
try {
  const results = await fetchProspects({ city, category, apiKey })
  console.log(JSON.stringify(results, null, 2))
} catch (err) {
  console.error(`Error fetching prospects: ${err.message}`)
  process.exit(1)
}
```

- Errors go to `console.error` (stderr), data goes to `console.log` (stdout) — allows piping
- Top-level `await` works because `"type": "module"` in package.json
